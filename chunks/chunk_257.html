SELECT
  t.trace_id, t.span_id, t.llm_call_id, t.prompt_template_version_id,
  t.completion_id, t.token_id, t.tool_call_id, t.agent_step_id,
  dp.profile_id, dp.column_id, dp.sketch_id, dp.drift_signature_id,
  de.drift_event_id, de.drift_type_id, de.reference_window_id,
  m.metric_id, m.metric_score_id, m.confidence_score_id,
  g.rule_id, g.trip_event_id, g.trip_payload_id, g.audit_log_id,
  lg.eval_run_id, lg.metric_id AS lg_metric_id, lg.llm_output_id,
  lg.judge_call_id, lg.judge_output_id,
  tid.tenant_id, pid.project_id, bid.billing_event_id
FROM whylabs.observed_traces t
JOIN whylabs.data_profiles dp ON dp.feature_id = t.feature_id
JOIN whylabs.drift_events de ON de.profile_id = dp.profile_id
JOIN whylabs.metrics m ON m.trace_id = t.trace_id
JOIN whylabs.guardrail_trips g ON g.llm_call_id = t.llm_call_id
JOIN whylabs.langkit_runs lg ON lg.llm_call_id = t.llm_call_id
WHERE tid.tenant_id = $tenant;
