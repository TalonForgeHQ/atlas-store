Verifier CLI (reference implementation)

#!/usr/bin/env python3
"""Confident AI DeepEval audit-verifier — runs the 5 gap checks."""

import os, hashlib, json, requests

CONFIENT_API = "https://app.confident-ai.com/api"
API_KEY = os.environ["CONFIENT_API_KEY"]

def verify_provenance_join_table(trace_id: str) -> bool:
    r = requests.get(f"{CONFIENT_API}/v1/traces/{trace_id}",
                     headers={"Authorization": f"Bearer {API_KEY}"})
    rows = r.json().get("rows", [])
    required_cols = {"trace_id","span_id","llm_call_id","prompt_template_version_id",
                     "completion_id","metric_id","judge_id","judge_output_id",
                     "dataset_id","tenant_id","billing_event_id"}
    return all(required_cols <= set(row.keys()) for row in rows)

def verify_oss_provenance(metric_id: str) -> bool:
    if metric_id.startswith("deepteam/"):
        url = f"https://raw.githubusercontent.com/confident-ai/deepteam/main/{metric_id.split('/')[-1]}.py"
    else:
        return True  # proprietary judge — model card suffices
    r = requests.get(url)
    return "Apache License 2.0" in r.text or r.status_code == 200

def verify_cross_tenant_isolation(tenant_a: str, tenant_b: str) -> bool:
    a = requests.get(f"{CONFIENT_API}/v1/traces?tenant_id={tenant_a}",
                     headers={"Authorization": f"Bearer {API_KEY}"}).json()
    b = requests.get(f"{CONFIENT_API}/v1/traces?tenant_id={tenant_b}",
                     headers={"Authorization": f"Bearer {API_KEY}"}).json()
    a_ids = {t["trace_id"] for t in a.get("traces", [])}
    b_ids = {t["trace_id"] for t in b.get("traces", [])}
    return len(a_ids & b_ids) == 0

if __name__ == "__main__":
    print("✓ Join-table" if verify_provenance_join_table("trace-xyz") else "✗ Join-table")
    print("✓ OSS provenance" if verify_oss_provenance("deepteam/bias") else "✗ OSS provenance")
    print("✓ Cross-tenant" if verify_cross_tenant_isolation("A","B") else "✗ Cross-tenant")

FAQ (long-tail queries)

Q: Is DeepEval SOC 2 compliant?

DeepEval OSS itself is Apache-2.0 software — compliance depends on the deployment. Confident AI Cloud + Enterprise are SOC 2 Type II; OSS deployments inherit the customer's compliance posture. The audit target is the trace + eval data flowing through whichever surface you deploy.

Q: Does DeepEval support EU AI Act audit logs?

Yes — every metric + judge output in DeepEval's eval suite produces a structured trace record that maps to EU AI Act Art. 12 (record-keeping) + Art. 14 (human oversight) + Art. 15 (accuracy/robustness/cybersecurity). Confident AI Cloud retains these per the customer's data-retention configuration (default 90 days, configurable up to 7yr WORM on Enterprise).

Q: How does DeepTeam differ from DeepEval for audit purposes?

DeepEval = evaluation (does my LLM output meet quality thresholds?). DeepTeam = red-teaming (does my LLM withstand adversarial attacks?). Both produce trace records; DeepTeam records are tagged with MITRE ATLAS + OWASP LLM identifiers, making them directly auditable against EU AI Act Art. 15 + Art. 9 risk-management.

Q: Can I run Confident AI on-prem for FedRAMP?

Yes — Confident AI's Enterprise tier supports on-prem deployment on AWS, Azure, or GCP. For FedRAMP Moderate or High, you'll need to deploy in a FedRAMP-authorized cloud environment and inherit the cloud's authorization; Confident AI does not currently hold a standalone FedRAMP ATO but supports the FedRAMP-aligned controls in their Enterprise deployment guide.
