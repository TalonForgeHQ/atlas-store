3. Audit gap #1 — End-to-end vector-mutation provenance join-table

Reference SQL join-table:

SELECT
  v.vector_id,
  v.namespace,
  v.tenant_id,
  v.operation,        -- 'upsert' | 'mutation' | 'delete'
  v.timestamp,
  v.api_key_id,
  v.api_key_rotation_event,
  v.rbac_role_change_event,
  v.sso_saml_assertion_id,
  v.audit_log_export_event_id,
  v.byok_key_rotation_event,
  v.cmek_key_rotation_event,
  v.embedding_call_id,    -- downstream OpenAI/Anthropic/Cohere/Hugging Face
  v.llm_call_id,          -- downstream LLM call
  v.warehouse_state_change_id,  -- downstream Snowflake/Databricks/BigQuery/Redshift/Postgres
  v.blob_state_change_id         -- downstream S3/GCS/Azure Blob
FROM pinecone.vector_audit_log v
LEFT JOIN pinecone.namespace_isolation n ON v.namespace = n.namespace
LEFT JOIN pinecone.tenant_isolation t ON v.tenant_id = t.tenant_id
LEFT JOIN pinecone.api_key_rotation a ON v.api_key_id = a.api_key_id
LEFT JOIN pinecone.rbac_role_change r ON v.rbac_role_change_event = r.event_id
LEFT JOIN pinecone.sso_saml_assertion s ON v.sso_saml_assertion_id = s.assertion_id
LEFT JOIN pinecone.audit_log_export e ON v.audit_log_export_event_id = e.event_id
LEFT JOIN pinecone.byok_rotation b ON v.byok_key_rotation_event = b.event_id
LEFT JOIN pinecone.cmek_rotation c ON v.cmek_key_rotation_event = c.event_id
LEFT JOIN downstream.openai_embeddings_call o ON v.embedding_call_id = o.call_id
LEFT JOIN downstream.llm_call l ON v.llm_call_id = l.call_id
LEFT JOIN downstream.warehouse_state_change w ON v.warehouse_state_change_id = w.event_id
LEFT JOIN downstream.blob_state_change bl ON v.blob_state_change_id = bl.event_id
WHERE v.timestamp BETWEEN :start AND :end
ORDER BY v.timestamp;
